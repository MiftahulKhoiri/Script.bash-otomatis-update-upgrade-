~$ apt install git

~$ git clone https://github.com/MiftahulKhoiri/Script.bash-otomatis-update-upgrade-.git

~$ cd /data/data/com.termux/files/home/Script.bash-otomatis-update-upgrade-/

~$ chmod +x update1.sh

~$ ./update1.sh

~$ ./update-script.sh
